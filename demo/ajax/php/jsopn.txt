public function ajax_add_cart_jsonp() {
		$callback 	= $_GET['callback'];
		$result 	= $this->add_cart();

		if (isset($result['code']) && isset($this->result_status[$result['code']])) {
			echo $callback . "(" . json_encode($this->result_status[$result['code']]) . ")";
		} else {
			echo $callback . "(" . json_encode(array('status' => 0, 'message' => 'ʧܣ')) . ")";
		}
	}
